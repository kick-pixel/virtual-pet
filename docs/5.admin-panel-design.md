```sql
SET NAMES utf8mb4;

  ADD COLUMN IF NOT EXISTS consecutive_days INT          NOT NULL DEFAULT 0
    COMMENT '当前连续签到天数（真实天数，不受7天奖励周期限制）' AFTER gen_disabled,
  ADD COLUMN IF NOT EXISTS max_streak_days  INT          NOT NULL DEFAULT 0
    COMMENT '历史最长连续签到天数'           AFTER consecutive_days;

-- ----------------------------------------------------------------
-- 2. ai_video_task 新增字段
-- ----------------------------------------------------------------
ALTER TABLE ai_video_task
  ADD COLUMN IF NOT EXISTS view_count    INT      NOT NULL DEFAULT 0
    COMMENT '浏览量'                     AFTER like_count,
  ADD COLUMN IF NOT EXISTS admin_status  TINYINT(1) NOT NULL DEFAULT 0
    COMMENT '管理状态 0-正常 1-封禁'    AFTER view_count;

-- 为视频管理封禁查询新增索引
-- 注意：MySQL 8.0 的 ALTER TABLE 不支持 ADD INDEX IF NOT EXISTS，改用 CREATE INDEX IF NOT EXISTS
CREATE INDEX IF NOT EXISTS idx_admin_status ON ai_video_task (admin_status);
CREATE INDEX IF NOT EXISTS idx_date_status  ON ai_video_task (create_time, status);

-- ----------------------------------------------------------------
-- 3. 统计中间表（用户生成次数汇总，避免每次全表聚合）
-- ----------------------------------------------------------------
CREATE TABLE IF NOT EXISTS virtual_user_stats (
  user_id          BIGINT   NOT NULL COMMENT '用户ID',
  total_gen_count  INT      NOT NULL DEFAULT 0 COMMENT '累计生成次数',
  fail_gen_count   INT      NOT NULL DEFAULT 0 COMMENT '生成失败次数',
  last_gen_time    DATETIME DEFAULT NULL COMMENT '最近生成时间',
  update_time      DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
                   ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (user_id),
  KEY idx_update_time (update_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
  COMMENT='用户生成统计中间表（每日定时刷新）';

-- ----------------------------------------------------------------
-- 4. 视频API每日统计中间表（按日期预聚合，加速 API 统计页面）
-- ----------------------------------------------------------------
CREATE TABLE IF NOT EXISTS virtual_api_daily_stats (
  stat_date       DATE     NOT NULL COMMENT '统计日期',
  total_count     INT      NOT NULL DEFAULT 0 COMMENT '调用总次数',
  success_count   INT      NOT NULL DEFAULT 0 COMMENT '成功次数',
  fail_count      INT      NOT NULL DEFAULT 0 COMMENT '失败次数',
  timeout_count   INT      NOT NULL DEFAULT 0 COMMENT '超时次数',
  avg_duration_s  DECIMAL(10,2) DEFAULT NULL COMMENT '平均生成耗时（秒）',
  success_rate    DECIMAL(5,2) DEFAULT NULL COMMENT '成功率（百分比）',
  create_time     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  update_time     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
                  ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (stat_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
  COMMENT='视频API每日统计中间表（定时任务每日凌晨刷新）';
```

### 2.3 统计中间表设计

#### 为什么需要中间表？

| 场景 | 直接查询问题 | 中间表方案 |
|------|------------|----------|
| 用户列表显示累计生成次数 | `ai_video_task` 可能百万行，每次 GROUP BY 全表扫描 | `virtual_user_stats` 预计算，列表 JOIN 即可 |
| API 统计 90 天趋势 | 每次 GROUP BY DATE + 多聚合函数，随数据增长越来越慢 | `virtual_api_daily_stats` 每日一条，查询极快 |

#### `virtual_user_stats` 刷新策略

```
方案A（推荐）：事件驱动增量更新
  - 视频任务完成/失败时，VideoCompletedListener/VideoFailedListener 中
    UPDATE virtual_user_stats SET total_gen_count = total_gen_count + 1 WHERE user_id = ?
  - 新用户注册时 INSERT INTO virtual_user_stats (user_id) VALUES (?)

方案B：定时全量刷新（每日凌晨 3:00）
  INSERT INTO virtual_user_stats (user_id, total_gen_count, fail_gen_count, last_gen_time)
  SELECT virtual_user_id,
         COUNT(*),
         SUM(CASE WHEN status='failed' THEN 1 ELSE 0 END),
         MAX(create_time)
  FROM ai_video_task
  WHERE virtual_user_id IS NOT NULL
  GROUP BY virtual_user_id
  ON DUPLICATE KEY UPDATE
    total_gen_count = VALUES(total_gen_count),
    fail_gen_count  = VALUES(fail_gen_count),
    last_gen_time   = VALUES(last_gen_time);
```

> 推荐方案A + 每日全量兜底（防止数据不一致）。

#### `virtual_api_daily_stats` 刷新策略

```sql
-- 定时任务每日凌晨 2:00 刷新昨日数据
INSERT INTO virtual_api_daily_stats
  (stat_date, total_count, success_count, fail_count, timeout_count, avg_duration_s, success_rate)
SELECT
  DATE(create_time)                                        AS stat_date,
  COUNT(*)                                                 AS total_count,
  SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END)   AS success_count,
  SUM(CASE WHEN status = 'failed' THEN 1 ELSE 0 END)      AS fail_count,
  SUM(CASE WHEN error_code = 'TIMEOUT' THEN 1 ELSE 0 END) AS timeout_count,
  ROUND(AVG(CASE WHEN status = 'completed' AND started_at IS NOT NULL
                 THEN TIMESTAMPDIFF(SECOND, started_at, completed_at) END), 2) AS avg_duration_s,
  ROUND(SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END)
        * 100.0 / NULLIF(COUNT(*), 0), 2)                                      AS success_rate
FROM ai_video_task
WHERE create_time >= DATE_SUB(CURDATE(), INTERVAL 1 DAY)
  AND create_time <  CURDATE()
GROUP BY DATE(create_time)
ON DUPLICATE KEY UPDATE
  total_count    = VALUES(total_count),
  success_count  = VALUES(success_count),
  fail_count     = VALUES(fail_count),
  timeout_count  = VALUES(timeout_count),
  avg_duration_s = VALUES(avg_duration_s),
  success_rate   = VALUES(success_rate),
  update_time    = NOW();
```

#### GET `/admin/virtual/checkin/list` — 分页查询用户签到统计

```sql

LEFT JOIN virtual_daily_checkin d ON d.user_id = u.user_id
<where>
  <if test="userId != null">AND u.user_id = #{userId}</if>
```

```java
// AdminVirtualApiStatsController.java
public AjaxResult getApiStats(String beginDate, String endDate) {
    // 1. 尝试从 virtual_api_daily_stats 读取预计算数据
    List<ApiDailyStats> stats = statsMapper.selectFromCache(beginDate, endDate);

    // 2. 对于今天（中间表未刷新）或中间表缺失的日期，实时聚合补充
    Set<LocalDate> cachedDates = stats.stream().map(s -> s.getStatDate()).collect(toSet());
    List<LocalDate> missingDates = getMissingDates(beginDate, endDate, cachedDates);
    if (!missingDates.isEmpty()) {
        List<ApiDailyStats> realtime = statsMapper.selectRealtimeStats(missingDates);
        stats.addAll(realtime);
    }

    // 3. 按日期排序返回
    stats.sort(Comparator.comparing(ApiDailyStats::getStatDate));
    return AjaxResult.success(stats);
}
```

**实时聚合 SQL**

```sql
SELECT
  DATE(create_time)                                        AS stat_date,
  COUNT(*)                                                 AS total_count,
  SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END)   AS success_count,
  SUM(CASE WHEN status = 'failed'    THEN 1 ELSE 0 END)   AS fail_count,
  SUM(CASE WHEN error_code = 'TIMEOUT' THEN 1 ELSE 0 END) AS timeout_count,
  ROUND(SUM(CASE WHEN status='completed' THEN 1 ELSE 0 END) * 100.0 / NULLIF(COUNT(*), 0), 2) AS success_rate,
  ROUND(AVG(CASE WHEN status = 'completed' AND started_at IS NOT NULL
                 THEN TIMESTAMPDIFF(SECOND, started_at, completed_at) END), 2)      AS avg_duration_s
FROM ai_video_task
WHERE create_time >= #{beginDate}
  AND create_time <  DATE_ADD(#{endDate}, INTERVAL 1 DAY)
GROUP BY DATE(create_time)
ORDER BY stat_date
```

## 5. 查询优化策略总结

### 5.1 索引设计

| 表 | 新增索引 | 用途 |
|---|---------|------|
| `ai_video_task` | `(create_time, status)` | API统计按日期+状态聚合 |
| `ai_video_task` | `(admin_status)` | 视频封禁状态过滤 |
| `virtual_user` | `(gen_disabled, status)` | 用户状态组合过滤 |
| `virtual_daily_checkin` | 已有 `idx_user_id` | 签到聚合无需新增 |

### 5.2 避免慢查询的核心原则

| 原则 | 说明 | 具体措施 |
|------|------|---------|
| **中间表预计算** | 高频聚合查询提前算好 | `virtual_user_stats`（生成次数）、`virtual_api_daily_stats`（API统计）|
| **限制 JOIN 深度** | 禁止超过 3 层关联 | 用户列表最多 3 个 LEFT JOIN（wallet + stats + credits）|
| **分页强制限制** | 禁止不带条件全表扫描 | `pageSize` 最大 100，后端强制 `startPage()`|
| **精确匹配优先** | 搜索条件用等值而非 LIKE | `userId`、`txHash`、`status` 均等值匹配 |
| **LIKE 加前缀索引** | 邮箱/钱包地址 LIKE 必须左匹配 | 拒绝 `%keyword%` 全模糊，提示用户填写前缀 |
| **今日数据实时，历史走缓存** | API统计页面 | 当日数据走 `ai_video_task` 实时聚合，其余日期走 `virtual_api_daily_stats`|

### 5.3 中间表刷新时序

```
每日 01:00 - virtual_user_stats 全量刷新（兜底）
每日 02:00 - virtual_api_daily_stats 刷新昨日数据
实时触发  - VideoCompletedListener 增量更新 virtual_user_stats
实时触发  - VirtualCheckinService 更新 max_streak_days
```

### 5.4 大数据量预案

当 `ai_video_task` 超过 500 万行时：

1. **分区表**：按月 RANGE 分区 `PARTITION BY RANGE(YEAR(create_time)*100+MONTH(create_time))`
2. **冷热分离**：3个月前的已完成任务归档到 `ai_video_task_archive`
3. **API统计彻底走中间表**：`virtual_api_daily_stats` 永久保留，不依赖原表

### 8.1 `virtual_user.register_ip` — 注册IP

**问题根因**：阅读 `VirtualAuthController` 发现所有登录入口均调用：

```java

//                                                      ↑ IP 传 null
```

**补齐策略**：区分"首次注册"和"再次登录"。注册时写入 `register_ip`，后续只更新 `last_login_ip`。

**获取客户端真实IP的工具方法**（放在 `VirtualAuthController` 或公共 `IpUtils`）：

```java
private String getClientIp(HttpServletRequest request) {
    String ip = request.getHeader("X-Forwarded-For");
    if (ip != null && !ip.isBlank() && !"unknown".equalsIgnoreCase(ip)) {
        return ip.split(",")[0].trim();   // 取第一个（真实客户端IP）
    }
    ip = request.getHeader("X-Real-IP");
    if (ip != null && !ip.isBlank()) return ip;
    return request.getRemoteAddr();
}
```

**需要修改的三个登录入口**（`VirtualAuthController.java`）：

| 入口方法 | 修改内容 |

| `emailLogin()` | 方法签名增加 `HttpServletRequest request`，将 `getClientIp(request)` 传给 `recordLoginSuccess` |
| `walletLogin()` | 同上 |
| `oauthCallback()` | 同上 |

**`IVirtualUserService.registerByEmail()` 修改**（首次注册时写入）：

### 8.4 `virtual_user.consecutive_days` 和 `max_streak_days` — 签到连续天数

**重要设计修正**：

`virtual_daily_checkin.day_sequence` 字段范围是 **1~7**，表示奖励周期的第几天，不代表真实连续天数。连续签到 10 天后 day_sequence = 3（10 mod 7），从 day_sequence 无法读出"连续10天"。

因此需要在 `virtual_user` 中增加 `consecutive_days`（真实连续天数）字段：

**DB 补充**（追加到 `9.admin_panel_20260301.sql`）：

```sql
ALTER TABLE virtual_user
  ADD COLUMN IF NOT EXISTS consecutive_days INT NOT NULL DEFAULT 0
    COMMENT '当前连续签到天数（真实天数，不受7天周期限制）' AFTER gen_disabled;
```

**签到 Service 更新逻辑**（`VirtualCheckinServiceImpl.checkin()`）：

```java
// 签到成功后，计算真实连续天数
LocalDate yesterday = LocalDate.now().minusDays(1);
VirtualDailyCheckin lastCheckin = checkinMapper.selectLatestByUserId(userId);

int newConsecutiveDays;
if (lastCheckin != null
    && lastCheckin.getCheckinDate().toLocalDate().equals(yesterday)) {
    // 昨天签到过 → 连续天数 +1
    newConsecutiveDays = user.getConsecutiveDays() + 1;
} else {
    // 断签或首次 → 重置为 1
    newConsecutiveDays = 1;
}
int newMaxStreak = Math.max(user.getMaxStreakDays(), newConsecutiveDays);
virtualUserMapper.updateStreakDays(userId, newConsecutiveDays, newMaxStreak);
```

**Mapper XML**：

```xml
<update id="updateStreakDays">
    UPDATE virtual_user
    SET consecutive_days = #{consecutiveDays},
        max_streak_days  = #{maxStreakDays},
        update_time      = NOW()
    WHERE user_id = #{userId}
</update>
```

**管理后台签到列表查询因此大幅简化**（无需子查询，直接读 `virtual_user` 字段）：

```sql
SELECT u.user_id, u.email,
       u.consecutive_days  AS current_streak,
       u.max_streak_days,
       MAX(d.checkin_date) AS last_checkin_date,
       COUNT(d.checkin_id) AS total_checkin_days
FROM virtual_user u
LEFT JOIN virtual_daily_checkin d ON d.user_id = u.user_id
GROUP BY u.user_id, u.email, u.consecutive_days, u.max_streak_days
ORDER BY last_checkin_date DESC
```

### 9.2 中间表与 `ai_video_task` 的字段映射关系

```
virtual_api_daily_stats 每行 = ai_video_task 某一天的聚合结果

stat_date      ←  DATE(ai_video_task.create_time)  GROUP BY
total_count    ←  COUNT(*)
success_count  ←  COUNT(*) WHERE status = 'completed'
fail_count     ←  COUNT(*) WHERE status = 'failed'
timeout_count  ←  COUNT(*) WHERE error_code = 'TIMEOUT'
                              OR error_message LIKE '%timeout%'
avg_duration_s ←  AVG(TIMESTAMPDIFF(SECOND, started_at, completed_at))
                  WHERE status='completed' AND started_at IS NOT NULL
success_rate   ←  success_count / NULLIF(total_count,0) * 100
```

**关键约束**：

- 中间表只存**已定型**的历史数据（昨天及之前）
- 今天的数据永远实时聚合（今天还未结束，数据持续变化）
- 每日定时任务用 `ON DUPLICATE KEY UPDATE` 覆盖刷新，保证幂等

`virtual_user_stats` 字段映射：

```
user_id         ←  ai_video_task.virtual_user_id  GROUP BY
total_gen_count ←  COUNT(*)
fail_gen_count  ←  COUNT(*) WHERE status = 'failed'
last_gen_time   ←  MAX(create_time)
```

### 9.3 定时任务设计

> **位置**：`ruoyi-admin` 模块（避免 `pet-virtual` 重复触发）
> **文件**：`ruoyi-admin/src/main/java/com/ruoyi/web/task/VirtualStatsRefreshTask.java`

```java
@Component
public class VirtualStatsRefreshTask {
    private static final Logger log = LoggerFactory.getLogger(VirtualStatsRefreshTask.class);

    @Autowired private VirtualApiDailyStatsMapper apiStatsMapper;
    @Autowired private VirtualUserStatsMapper     userStatsMapper;

    /** 每日凌晨 1:00 全量刷新 virtual_user_stats（兜底，防止事件遗漏） */
    @Scheduled(cron = "0 0 1 * * ?")
    @DistributedLock(key = "schedule:refresh:user:stats", leaseTime = 600)
    public void refreshUserStats() {
        log.info("[VirtualStats] 全量刷新 virtual_user_stats 开始");
        int rows = userStatsMapper.upsertAllFromAiVideoTask();
        log.info("[VirtualStats] virtual_user_stats 刷新完成，影响行数: {}", rows);
    }

    /** 每日凌晨 2:00 刷新昨日 API 统计 */
    @Scheduled(cron = "0 0 2 * * ?")
    @DistributedLock(key = "schedule:refresh:api:stats", leaseTime = 300)
    public void refreshApiDailyStats() {
        String yesterday = LocalDate.now().minusDays(1).toString();
        log.info("[VirtualStats] 刷新 API 统计日期: {}", yesterday);
        int rows = apiStatsMapper.upsertDailyStats(yesterday);
        log.info("[VirtualStats] virtual_api_daily_stats 刷新完成，日期: {}, 行数: {}", yesterday, rows);
    }

    /** 每小时半点刷新今日中间表（减少 Admin 页面实时查询压力） */
    @Scheduled(cron = "0 30 * * * ?")
    @DistributedLock(key = "schedule:refresh:api:today", leaseTime = 60)
    public void refreshTodayApiStats() {
        String today = LocalDate.now().toString();
        apiStatsMapper.upsertDailyStats(today);   // 今日数据覆盖写
    }
}
```

**`VirtualApiDailyStatsMapper.xml` 核心 SQL**：

```xml
<insert id="upsertDailyStats">
    INSERT INTO virtual_api_daily_stats
        (stat_date, total_count, success_count, fail_count,
         timeout_count, avg_duration_s, success_rate)
    SELECT
        #{statDate},
        COUNT(*)                                                             AS total_count,
        SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END)               AS success_count,
        SUM(CASE WHEN status = 'failed'    THEN 1 ELSE 0 END)               AS fail_count,
        SUM(CASE WHEN error_code = 'TIMEOUT'
                  OR error_message LIKE '%timeout%' THEN 1 ELSE 0 END)       AS timeout_count,
        ROUND(AVG(
            CASE WHEN status = 'completed'
                  AND started_at  IS NOT NULL
                  AND completed_at IS NOT NULL
                 THEN TIMESTAMPDIFF(SECOND, started_at, completed_at)
            END
        ), 2)                                                                AS avg_duration_s,
        ROUND(
            SUM(CASE WHEN status='completed' THEN 1 ELSE 0 END)
            * 100.0 / NULLIF(COUNT(*), 0),
        2)                                                                   AS success_rate
    FROM ai_video_task
    WHERE create_time >= #{statDate}
      AND create_time  &lt; DATE_ADD(#{statDate}, INTERVAL 1 DAY)
    ON DUPLICATE KEY UPDATE
        total_count    = VALUES(total_count),
        success_count  = VALUES(success_count),
        fail_count     = VALUES(fail_count),
        timeout_count  = VALUES(timeout_count),
        avg_duration_s = VALUES(avg_duration_s),
        success_rate   = VALUES(success_rate),
        update_time    = NOW()
</insert>
```

**`VirtualUserStatsMapper.xml` 全量刷新 SQL**：


### 9.4 Admin API 混合查询逻辑

```java
// AdminVirtualApiStatsController.java
@GetMapping("/api")
@PreAuthorize("@ss.hasPermi('virtual:stats:list')")
public AjaxResult getApiStats(
        @RequestParam(defaultValue = "7")  int    days,
        @RequestParam(required = false)    String beginDate,
        @RequestParam(required = false)    String endDate) {

    LocalDate end   = endDate   != null ? LocalDate.parse(endDate)   : LocalDate.now();
    LocalDate begin = beginDate != null ? LocalDate.parse(beginDate) : end.minusDays(days - 1);
    LocalDate today = LocalDate.now();

    List<VirtualApiDailyStats> result = new ArrayList<>();

    // ① 历史日期：读中间表，O(days) 极快
    if (begin.isBefore(today)) {
        LocalDate histEnd = end.isBefore(today) ? end : today.minusDays(1);
        result.addAll(apiStatsMapper.selectByDateRange(
                begin.toString(), histEnd.toString()));
    }

    // ② 今日数据：实时聚合（或读每小时刷新的中间表缓存，见 9.3 每小时任务）
    if (!end.isBefore(today)) {
        VirtualApiDailyStats todayStats =
                apiStatsMapper.selectRealtimeForDate(today.toString());
        if (todayStats != null) result.add(todayStats);
    }

    result.sort(Comparator.comparing(VirtualApiDailyStats::getStatDate));
    return AjaxResult.success(result);
}
```

```
         SUM(CASE WHEN status='failed' THEN 1 ELSE 0 END), MAX(create_time)
  FROM ai_video_task WHERE virtual_user_id IS NOT NULL GROUP BY virtual_user_id
  ON DUPLICATE KEY UPDATE total_gen_count=VALUES(total_gen_count),
```
